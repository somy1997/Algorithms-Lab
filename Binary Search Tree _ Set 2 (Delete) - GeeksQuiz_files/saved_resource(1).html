                                            tree *temp=new tree();

                                            temp->right=NULL;
                                            temp->parent=NULL;

                                            tree *insertkey(tree *parent ,tree *root,int key)
                                            {
                                            if(root==NULL)
                                            {
                                            //in that case create a new node andd return a pointer to it ;
                                            tree *temp=createnode(key);
                                            temp->parent=parent;
                                            return temp;
                                            }
                                            //if key is less or equal to current root vlaue recurse on the left
                                            if(key<=root->data)
                                            root->left=insertkey(root,root->left,key);
                                            else
                                            root->right=insertkey(root,root->right,key);
                                            return root ;
                                            }
                                            //to return the pointer to the predecessor of the given key in the tree
                                            //if left present then return the poiner to it ;
                                            //if NULL go till parent whose value is less than that of the given value
                                            tree *predecessor(tree *root ,int key)
                                            {
                                            //first search the key if found then return the parent pointer ;
                                            if(root==NULL)
                                            return NULL ;
                                            if(root->data==key)
                                            {
                                            if(root->left!=NULL)
                                            {
                                            //in that case move to the left child and go to the as right as possible;
                                            tree *temp=root->left;
                                            while(temp->right!=NULL)
                                            temp=temp->right;
                                            return temp;
                                            }
                                            else
                                            //go upward till the parent of the given node till the value is less is not
                                            {
                                            tree *parent=root->parent;
                                            while(parent!=NULL)
                                            { if(parent->data<root->data)
                                            break;
                                            parent=parent->parent;
                                            }
                                            return parent ;
                                            }
                                            }
                                            tree *temp1;
                                            //otherwise recurse to the left or right of the tree;
                                            if(key<root->data)
                                            temp1= predecessor(root->left,key);
                                            else
                                            temp1= predecessor(root->right,key);
                                            return temp1;
                                            }

                                            void swappointers(tree *temp1,tree *temp2)
